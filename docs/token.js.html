'use strict';

// Dependencies
var mongoose = require('mongoose');
var hash = require('../util/hash');

/**
 * Callback to use for asynchronous calls into the token object.
 * @callback Token#tokenCallback
 * @param {Token#tokenError} err - error object if an error occurs, null otherwise
 * @param {Object} token - token object
 **/

/**
 * Error object passed to the done callback when an error occurs in this module.
 * @typedef {Object} Token#tokenError
 * @property {integer} code - Error code for the error that was generated.
 * @property {String} type - One of 'error', 'warning', 'info'.
 * @property {String} message - Detailed error message.
 **/

/**
 * Data to associate with a particular token.
 * @typedef {Object} Token#tokenData
 * @property {String} type - Type to associate with the token. All similar usage should use the same token type.
 * @property {String} value - Associated value for this particular token. For example, a userid for further lookup.
 */

/**
 * The token class implements one-time persistent objects with associated data.
 * XXX We will want to occasionally go through the Token table in the database and delete old tokens that haven't been used.
 * XXX Implement a clean up method.
 * @constructor Token
 * @property {String} id - Internal use only.
 * @property {String} type - Type to associate with the token. All similar usage should use the same token type.
 * @property {String} value - Associated value for this particular token. For example, a userid for further lookup.
 * @property {Date} created=now - Internal use only.
 **/
var Token = new mongoose.Schema({
    id: String,
    type: String,
    value: String,
    created: {
        type: Date,
        default: Date.now
    }
});

/**
 * Consume a generated token
 * @method Token.consume
 * @param {String}   tokid - Token id previously generated by {@link Token#generate}
 * @param {Token#tokenCallback} done  - Callback to be called when the method is complete
 * @return Returns the result of the calling the done function back to the caller
 */
Token.statics.consume = function (tokid, done) {
    var Token = this;
    Token.findOne({
        id: tokid
    }, function (err, token) {
        if (err) {
            return done(err, token);
        }
        if (!token) {
            return done({
                code: 404,
                type: 'error',
                message: 'Token not found.'
            });
        }

        token.remove(); // One time use.

        return done(null, token);
    });
};

/** Generate a persistent token for one-time usage.
 * @method Token.generate
 * @param {Token#tokenData} data - Token data to save including type and value.
 * @param {Token#tokenCallback} done - Callback to be called when the method is complete.
 * @param {integer} [keylen=128] - Key length for the token to be generated
 * @return Returns the result of the calling the done function back to the caller
 **/
Token.statics.generate = function (data, done, keylen) {
    var Token = this;
    data.id = hash.createSalt(keylen);
    Token.create(data, function (err, token) {
        if (err) {
            return done(err, token);
        }
        return done(null, token);
    });
};

var t = mongoose.model('Token', Token);

module.exports = t;
